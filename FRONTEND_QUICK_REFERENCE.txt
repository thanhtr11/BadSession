MISTAKE 1: Inline functions in render
  WRONG: <button onClick={() => deleteItem(id)}>Delete</button>
  RIGHT: 
    const handleDelete = useCallback(() => {
      deleteItem(id);
    }, [id]);
    <button onClick={handleDelete}>Delete</button>

MISTAKE 2: Complex state in single useState
  WRONG: const [name, setName] = useState('');
         const [email, setEmail] = useState('');
         const [phone, setPhone] = useState('');
  RIGHT: const [form, setForm] = useState({
           name: '', email: '', phone: ''
         });

MISTAKE 3: No PropTypes
  WRONG: function Component(props) { ... }
  RIGHT: Component.propTypes = { 
           id: PropTypes.number.isRequired,
           name: PropTypes.string
         };

MISTAKE 4: Async operation without dependency array
  WRONG: useEffect(() => { fetchData(); });
  RIGHT: useEffect(() => { fetchData(); }, [dependency]);

MISTAKE 5: Rendering large lists without memo
  WRONG: {items.map(i => <Item item={i} />)}
  RIGHT: const Item = memo(({item}) => ...);
         {items.map(i => <Item key={i.id} item={i} />)}

--- Functional Component with PropTypes ---

import { memo, useCallback } from 'react';
import PropTypes from 'prop-types';

const MyComponent = memo(function MyComponent({ data, onAction }) {
  const handleClick = useCallback(() => {
    onAction(data.id);
  }, [data.id, onAction]);

  return (
    <button onClick={handleClick}>
      {data.name}
    </button>
  );
});

MyComponent.propTypes = {
  data: PropTypes.shape({
    id: PropTypes.number.isRequired,
    name: PropTypes.string.isRequired
  }).isRequired,
  onAction: PropTypes.func
};

MyComponent.defaultProps = {
  onAction: () => {}
};

export default MyComponent;

--- Custom Hook Template ---

import { useState, useEffect, useCallback } from 'react';

export const useMyHook = (initialValue) => {
  const [value, setValue] = useState(initialValue);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState(null);

  const fetch = useCallback(async () => {
    setLoading(true);
    try {
      const data = await fetchData();
      setValue(data);
    } catch (err) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    fetch();
  }, [fetch]);

  return { value, loading, error, refetch: fetch };
};
